DECLARE @ProviderNumber INT = 10007928
DECLARE @AcademicYear NVARCHAR(5) = '23/24'


SELECT
	FD.LearningAimCode,
	FD.LearningAimTitle,
	FD.FundLineType,
	FD.FundLineSummary,
	FD.LearnRefNumber,
	FD.AimSeqNumber,
	FD.AcademicYear,
	FD.ILRReturn,
	FD.LatestImportedFileAcademicYear,
	FD.LatestImportedFileILRReturn,
	FD.StartDate,
	FD.PlannedEndDate,
	FD.ActualEndDate,
	FD.IsFundedStart,
	FD.IsOverallStart,
	FD.IsOverallLeaver,
	FD.IsOverallContinuerFunded,
	FD.IsOverallCompletedFunded,
	FD.IsOverallRetainedFunded,
	FD.IsOverallWithdrawnFunded,
	FD.IsOverallAchieverFunded,
	FD.FAMLearningDeliveryLDM
FROM FIS_FundingData FD
WHERE
	FD.ProviderNumber = @ProviderNumber
	AND FD.HybridEndYear = @AcademicYear
	--AND FD.IncludeInQARMeasures = 1
	AND FD.LearningAimCode = '40009488'
	AND FD.FundLineType = 'ESFA'
	AND FD.IsOverallStart = 1
	--AND FD.IsOverallLeaver = 1
	--AND FD.IsOverallCompletedFunded = 1
	--AND FD.IsOverallAchieverFunded = 1
	--AND FD.IsOverallContinuerFunded = 1
	--AND FD.LearnRefNumber = 'F253521'
ORDER BY
	FD.LearningAimCode,
	FD.LearnRefNumber
