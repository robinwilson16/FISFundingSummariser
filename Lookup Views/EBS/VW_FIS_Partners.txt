CREATE VIEW VW_FIS_Partners
AS
	SELECT
		PartnerUKPRN = ORG.ORGANISATION_CODE,
		PartnerName = MIN ( ORG.FES_FULL_NAME )
	FROM [EBS-DB\EBS].ebs.dbo.ORGANISATION_UNITS ORG
	WHERE
		ORG.FES_ACTIVE = 'Y'
		AND ISNUMERIC ( ORG.ORGANISATION_CODE ) = 1
		AND LEN ( LTRIM ( RTRIM ( ORG.ORGANISATION_CODE ) ) ) = 8
	GROUP BY
		ORG.ORGANISATION_CODE
