CREATE VIEW VW_FIS_Employers
AS
	SELECT
		EmployerEDRS = ORG.ORGANISATION_CODE,
		EmployerName = MIN ( ORG.FES_FULL_NAME )
	FROM [EBS-DB\EBS].ebs.dbo.ORGANISATION_UNITS ORG
	WHERE
		ORG.FES_ACTIVE = 'Y'
		AND ISNUMERIC ( ORG.ORGANISATION_CODE ) = 1
		AND LEN ( LTRIM ( RTRIM ( ORG.ORGANISATION_CODE ) ) ) = 9
	GROUP BY
		ORG.ORGANISATION_CODE
