CREATE VIEW [dbo].[FIS_CollegeStructure]
AS
	SELECT
		AcademicYear,
		ProvSpecValue,
		CourseCode,
		CourseTitle,
		CollegeLevel1Code,
		CollegeLevel1Name,
		CollegeLevel2Code,
		CollegeLevel2Name,
		CollegeLevel3Code,
		CollegeLevel3Name,
		CollegeLevel4Code,
		CollegeLevel4Name
	FROM OPENQUERY([IEG-MISSQL-01], '
	SET FMTONLY OFF; 
	EXEC IEG.dbo.SPR_FIS_CollegeStructure
		WITH RESULT SETS
		(
			(
				AcademicYear CHAR(5),
				ProvSpecValue VARCHAR(255),
				CourseCode VARCHAR(255),
				CourseTitle VARCHAR(255),
				CollegeLevel1Code VARCHAR(12),
				CollegeLevel1Name VARCHAR(150),
				CollegeLevel2Code VARCHAR(12),
				CollegeLevel2Name VARCHAR(150),
				CollegeLevel3Code VARCHAR(12),
				CollegeLevel3Name VARCHAR(150),
				CollegeLevel4Code VARCHAR(12),
				CollegeLevel4Name VARCHAR(150)
			)
		);
	') X
GO
